jQuery

var bannerSlide=new Swiper('.banner-slider', {

 loop: false,

 simulateTouch:false,

 slidesPerView: 5,  //슬라이드 보여지는 개수

 spaceBetween: 25,  //슬라이드 간격

 speed:600,

 autoplay: {

  delay: 3000,

  disableOnInteraction: false

 },

 navigation: {

  nextEl: '.banner-slider .swiper-button-next',

  prevEl: '.banner-slider .swiper-button-prev'

 },

 breakpoints: {

  1023: {  //슬라이드 변화 시점 1023px

   slidesPerView: 3,

   spaceBetween: 25

  },

  767: {  //슬라이드 변화 시점 767px

   slidesPerView: 1

  }

 }

});


//banner slide play/stop

var playSild5=true;

$('.banner-slider .control-slide').on('click',function(){

 if(playSild5==true){

  $(this).addClass('on');

  bannerSlide.autoplay.stop();

 } else {

  $(this).removeClass('on');

  bannerSlide.autoplay.start();

 }

 playSild5 = ! playSild5;

});

  function loadSlide() {

   var controlobj = {};

   controlobj.autoMove = true;   //슬라이드 자동재생 true or false

   controlobj.hoverStop = false;   //autoMove = true일 경우, 마우스 오버시 stop 이벤트 true or false

   controlobj.time = 5000;   //슬라이드 재생시간

   controlobj.transition = 500;   //슬라이드 넘어가는 시간


   slide.set(controlobj);

  }

  loadSlide();

 }

});
